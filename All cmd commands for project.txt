# ============================================================
# SHOPPERS INTENT MLOPS — SETUP COMMANDS GUIDE
# ============================================================

# -----------------------------------------------
# STEP 1 — CREATE PROJECT DIRECTORY
# -----------------------------------------------
mkdir shoppers-intent-mlops
cd shoppers-intent-mlops

# -----------------------------------------------
# STEP 2 — CREATE VIRTUAL ENVIRONMENT
# -----------------------------------------------
python -m venv venv

# activate virtual environment (Windows)
venv\Scripts\activate

# activate virtual environment (Mac/Linux)
# source venv/bin/activate

# -----------------------------------------------
# STEP 3 — CREATE PROJECT FOLDER STRUCTURE
# -----------------------------------------------
mkdir api
mkdir api\templates
mkdir api\static
mkdir data
mkdir etl
mkdir ml
mkdir monitoring
mkdir scheduler
mkdir tests
mkdir utils
mkdir logs
mkdir .github
mkdir .github\workflows

# create empty __init__.py files for python packages
type nul > api\__init__.py
type nul > etl\__init__.py
type nul > ml\__init__.py
type nul > scheduler\__init__.py
type nul > tests\__init__.py
type nul > utils\__init__.py

# -----------------------------------------------
# STEP 4 — CREATE ALL PROJECT FILES
# -----------------------------------------------
# create root level files
type nul > .env
type nul > .env.example
type nul > .gitignore
type nul > .dockerignore
type nul > Dockerfile
type nul > render.yaml
type nul > requirements.txt
type nul > README.md

# create utils files
type nul > utils\logger.py

# create etl files
type nul > etl\extract.py
type nul > etl\transform.py
type nul > etl\load.py

# create data files
type nul > data\reload_data.py
# NOTE: manually copy online_shoppers_intention.csv into data/ folder

# create ml files
type nul > ml\preprocess.py
type nul > ml\train.py
type nul > ml\evaluate.py
type nul > ml\predict.py
type nul > ml\mlflow_tracker.py

# create monitoring files
type nul > monitoring\drift_report.py

# create scheduler files
type nul > scheduler\retrain_job.py

# create api files
type nul > api\main.py
type nul > api\templates\index.html
type nul > api\static\style.css
type nul > api\static\script.js

# create test files
type nul > tests\test_api.py

# create ci/cd file
type nul > .github\workflows\ci_cd.yml

# -----------------------------------------------
# STEP 5 — INSTALL DEPENDENCIES
# -----------------------------------------------
pip install -r requirements.txt

# -----------------------------------------------
# STEP 6 — MYSQL SETUP (MySQL Workbench)
# -----------------------------------------------
# run this in MySQL Workbench:
# CREATE DATABASE shoppers_db;

# -----------------------------------------------
# STEP 7 — POSTGRESQL SETUP (pgAdmin)
# -----------------------------------------------
# run this in pgAdmin Query Tool:
# CREATE DATABASE shoppers_warehouse;

# -----------------------------------------------
# STEP 8 — LOAD CSV DATA INTO MYSQL
# -----------------------------------------------
python data/reload_data.py

# -----------------------------------------------
# STEP 9 — RUN ETL PIPELINE (MySQL → PostgreSQL)
# -----------------------------------------------
python etl/load.py

# -----------------------------------------------
# STEP 10 — TRAIN ML MODEL
# -----------------------------------------------
python ml/preprocess.py

python ml/train.py

# -----------------------------------------------
# STEP 11 — EVALUATE MODEL
# -----------------------------------------------
python ml/evaluate.py

# -----------------------------------------------
# STEP 12 — TEST PREDICTION
# -----------------------------------------------
python ml/predict.py

# -----------------------------------------------
# STEP 13 — GENERATE DRIFT REPORT
# -----------------------------------------------
python monitoring/drift_report.py

# -----------------------------------------------
# STEP 14 — RUN API LOCALLY
# -----------------------------------------------
uvicorn api.main:app --reload
# open browser: http://127.0.0.1:8000
# swagger ui: http://127.0.0.1:8000/docs

# -----------------------------------------------
# STEP 15 — RUN TESTS
# -----------------------------------------------
pytest tests/test_api.py -v

# -----------------------------------------------
# STEP 16 — RUN SCHEDULER MANUALLY
# -----------------------------------------------
python -c "from scheduler.retrain_job import run_pipeline; run_pipeline()"

# -----------------------------------------------
# STEP 17 — VIEW MLFLOW UI
# -----------------------------------------------
	run in cmd
	mlflow ui
# open browser: http://127.0.0.1:5000

# STEP 18 — BUILD DOCKER IMAGE
# -----------------------------------------------
	in cmd 
	docker --version
	where docker
	docker ps
	wsl --update
docker build -t shoppers-intent-mlops .

if not run use below command

docker run -p 8001:8000 --env-file .env shoppers-intent-mlops
